int poz(int x1, int y1, int l1, int  i1, int  x2, int y2, int  l2, int i2) {
	struct punct {
		int x, y;
		bool equals(punct a) {
			if (x == a.x && a.y == y)
				return true;
			return false;
		}
	};

	struct dreptunghi {
		punct a, b;
		dreptunghi(int x, int y, int lungime, int inaltime) {
			a.x = x;
			a.y = y;
			b.x = a.x + lungime;
			b.y = a.y + inaltime;
		}
		bool contine(punct z)
		{
			if (a.x < z.x && b.x > z.x && a.y < z.y && b.y > z.y)
				return true;
			return false;
		}
		bool equals(dreptunghi d) {
			if (a.equals(d.a) && b.equals(d.b) || a.equals(d.b) && b.equals(d.a))
				return true;
			return false;
		}
		bool contineDreptunghi(dreptunghi d) {
			if (contine(d.a) && contine(d.b))
				return true;
			return false;
		}
	};

	dreptunghi a(x1, y1, l1, i1);
	dreptunghi b(x2, y2, l2, i2);

	if (a.equals(b))
		return 1;
	if (a.contineDreptunghi(b) || b.contineDreptunghi(a))
		return 0;
	if (a.contine(b.a) || a.contine(b.b) || b.contine(a.a) || b.contine(a.b))
		return 3;
	return 2;
}
